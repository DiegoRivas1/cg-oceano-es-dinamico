\begin{abstract}
Este informe describe el diseño e implementación de un sistema de animación
por computador que simula la superficie de un océano en movimiento mediante
OpenGL. Se detalla la generación de la malla, el cálculo de alturas basado en
la superposición de olas sinusoidales, el cómputo de normales, el
texturizado, la iluminación y los elementos adicionales incorporados a la
escena.
\end{abstract}

\section{Modelo Matemático de las Olas}
% Reproducir la fórmula del enunciado:
\begin{equation}
    h(x, z, t) = \sum_{i=1}^{n} A_i \cos\big(k_i (x\cos(d_i) + z\sin(d_i)) - 2\pi f_i t + p_i\big)
\end{equation}
\begin{equation}
    k_i = \frac{4\pi^2 f_i^2}{9.81}
\end{equation}
% Explicar de dónde salen los parámetros: archivo data/spectrum.txt (19
% olas, columnas amplitud/dirección/frecuencia) y que la fase se generó
% aleatoriamente con semilla fija para reproducibilidad.

\subsection{Cálculo en GPU vs. CPU}
% Justificar la decisión: 19 olas x malla de ~150x150 puntos x 60fps es
% mucho para hacerlo en CPU cada frame; se mueve el cómputo al vertex
% shader. Mostrar fragmento relevante de oceano.vert.
\begin{lstlisting}[language=C, caption={Fragmento de oceano.vert: cálculo de altura y normal analítica}]
for (int i = 0; i < numOlas; ++i) {
    float faseTotal = numeroOnda[i] * (x*dirX + z*dirZ)
                     - DOS_PI * frecuencia[i] * tiempo + fase[i];
    altura += amplitud[i] * cos(faseTotal);
    float derivComun = -amplitud[i] * numeroOnda[i] * sin(faseTotal);
    dHdx += derivComun * dirX;
    dHdz += derivComun * dirZ;
}
normal = normalize(vec3(-dHdx, 1.0, -dHdz));
\end{lstlisting}
